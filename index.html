const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const multer = require('multer'); const path = require('path'); const app = express(); const server = http.createServer(app); const io = socketIo(server); const APP_NAME = "Tambayan sa Space"; let users = []; // Store user locations // Set up storage for uploaded images const storage = multer.diskStorage({ destination: './uploads/', filename: (req, file, cb) => { cb(null, `${Date.now()}_${file.originalname}`); } }); const upload = multer({ storage }); app.use('/uploads', express.static(path.join(__dirname, 'uploads'))); io.on('connection', (socket) => { console.log(`${APP_NAME}: A user connected`); socket.on('setLocation', (location) => { users[socket.id] = location; console.log(`${APP_NAME}: User ${socket.id} set location:`, location); }); socket.on('sendMessage', (data) => { const senderLocation = users[socket.id]; if (!senderLocation) return; for (let [id, location] of Object.entries(users)) { if (id !== socket.id && getDistance(senderLocation, location) < 10) { io.to(id).emit('receiveMessage', data); } } }); socket.on('disconnect', () => { delete users[socket.id]; console.log(`${APP_NAME}: A user disconnected`); }); }); // Handle image uploads app.post('/upload', upload.single('image'), (req, res) => { if (!req.file) { return res.status(400).json({ error: 'No file uploaded' }); } res.json({ imageUrl: `/uploads/${req.file.filename}` }); }); function getDistance(loc1, loc2) { const R = 6371; // Radius of the Earth in km const dLat = (loc2.lat - loc1.lat) * (Math.PI / 180); const dLon = (loc2.lon - loc1.lon) * (Math.PI / 180); const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(loc1.lat * (Math.PI / 180)) * Math.cos(loc2.lat * (Math.PI / 180)) * Math.sin(dLon/2) * Math.sin(dLon/2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); return R * c; // Distance in km } server.listen(3000, () => { console.log(`${APP_NAME} Server is running on port 3000`); });
